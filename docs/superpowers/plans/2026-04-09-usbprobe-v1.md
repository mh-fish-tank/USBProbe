# USBProbe v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Linux desktop application for real-time USB device monitoring with full descriptor inspection, hex viewing, event logging, and udev rules management.

**Architecture:** Electron app with React+TypeScript renderer, Rust native addon (napi-rs) running in a Node.js worker thread for USB device access via sysfs/udev. SQLite for persistent event storage. Catppuccin dark theme with Iconify icons.

**Tech Stack:** Electron, React 18, TypeScript, Vite (electron-vite), Rust (napi-rs), SQLite (better-sqlite3), Zustand, react-i18next, Iconify, Vitest

```rust
fn read_sysfs_attr(base: &Path, attr: &str) -> Option<String> {
  fs::read_to_string(base.join(attr))
    .ok()
    .map(|s| s.trim().to_string())
}

fn read_sysfs_u32(base: &Path, attr: &str) -> u32 {
  read_sysfs_attr(base, attr)
    .and_then(|s| {
      if s.starts_with("0x") || s.starts_with("0X") {
        u32::from_str_radix(&s[2..], 16).ok()
      } else {
        s.parse().ok()
      }
    })
    .unwrap_or(0)
}

fn read_sysfs_hex(base: &Path, attr: &str) -> u32 {
  read_sysfs_attr(base, attr)
    .and_then(|s| {
      let s = s.trim_start_matches("0x").trim_start_matches("0X");
      u32::from_str_radix(s, 16).ok()
    })
    .unwrap_or(0)
}

fn parse_bcd(val: u32) -> String {
  format!("{}.{:02}", (val >> 8) & 0xFF, val & 0xFF)
}

fn parse_speed(speed_str: &str) -> (String, f64) {
  match speed_str {
    "1.5" => ("low".into(), 1.5),
    "12" => ("full".into(), 12.0),
    "480" => ("high".into(), 480.0),
    "5000" => ("super".into(), 5000.0),
    "10000" | "20000" => ("super_plus".into(), speed_str.parse().unwrap_or(10000.0)),
    _ => ("full".into(), 12.0),
  }
}

fn parse_transfer_type(attrs: u32) -> String {
  match attrs & 0x03 {
    0 => "control".into(),
    1 => "isochronous".into(),
    2 => "bulk".into(),
    3 => "interrupt".into(),
    _ => "control".into(),
  }
}

fn interface_class_name(class: u32, subclass: u32) -> String {
  match class {
    0x01 => "Audio".into(),
    0x02 => "CDC Communications".into(),
    0x03 => match subclass {
      0x01 => "HID — Boot Interface".into(),
      _ => "HID".into(),
    },
    0x05 => "Physical".into(),
    0x06 => "Image".into(),
    0x07 => "Printer".into(),
    0x08 => "Mass Storage".into(),
    0x09 => "Hub".into(),
    0x0A => "CDC Data".into(),
    0x0B => "Smart Card".into(),
    0x0D => "Content Security".into(),
    0x0E => "Video".into(),
    0x0F => "Personal Healthcare".into(),
    0x10 => "Audio/Video".into(),
    0xDC => "Diagnostic".into(),
    0xE0 => "Wireless Controller".into(),
    0xEF => "Miscellaneous".into(),
    0xFE => "Application Specific".into(),
    0xFF => "Vendor Specific".into(),
    _ => format!("Unknown (0x{:02X})", class),
  }
}

fn read_raw_descriptor(base: &Path) -> Vec<u32> {
  let desc_path = base.join("descriptors");
  match fs::read(&desc_path) {
    Ok(bytes) => bytes.iter().map(|b| *b as u32).collect(),
    Err(_) => vec![],
  }
}

fn parse_port_numbers(sysfs_path: &str) -> Vec<u32> {
  // sysfs path like "1-2.3.1" → port numbers [2, 3, 1]
  let basename = Path::new(sysfs_path)
    .file_name()
    .and_then(|s| s.to_str())
    .unwrap_or("");

  if let Some(ports_part) = basename.split('-').nth(1) {
    ports_part
      .split('.')
      .filter_map(|p| p.parse::<u32>().ok())
      .collect()
  } else {
    vec![]
  }
}

fn read_endpoints(iface_path: &Path) -> Vec<NativeEndpoint> {
  let mut endpoints = vec![];
  let entries = match fs::read_dir(iface_path) {
    Ok(e) => e,
    Err(_) => return endpoints,
  };

  for entry in entries.flatten() {
    let name = entry.file_name().to_string_lossy().to_string();
    if name.starts_with("ep_") {
      let ep_path = entry.path();
      let address = read_sysfs_hex(&ep_path, "bEndpointAddress");
      let attrs = read_sysfs_hex(&ep_path, "bmAttributes");
      let direction = if read_sysfs_attr(&ep_path, "direction").as_deref() == Some("in") {
        "in"
      } else {
        "out"
      };

      endpoints.push(NativeEndpoint {
        b_endpoint_address: address,
        direction: direction.into(),
        bm_attributes: attrs,
        transfer_type: parse_transfer_type(attrs),
        w_max_packet_size: read_sysfs_u32(&ep_path, "wMaxPacketSize"),
        b_interval: read_sysfs_u32(&ep_path, "bInterval"),
        raw_descriptor: vec![],
      });
    }
  }
  endpoints
}

fn read_interfaces(config_path: &Path, bus: u32) -> Vec<NativeInterface> {
  let mut interfaces = vec![];
  let entries = match fs::read_dir(config_path) {
    Ok(e) => e,
    Err(_) => return interfaces,
  };

  for entry in entries.flatten() {
    let name = entry.file_name().to_string_lossy().to_string();
    // Interface dirs look like "1-2:1.0" (bus-port:config.interface)
    if name.contains(':') {
      let iface_path = entry.path();
      let class = read_sysfs_hex(&iface_path, "bInterfaceClass");
      let subclass = read_sysfs_hex(&iface_path, "bInterfaceSubClass");

      interfaces.push(NativeInterface {
        b_interface_number: read_sysfs_hex(&iface_path, "bInterfaceNumber"),
        b_alternate_setting: read_sysfs_hex(&iface_path, "bAlternateSetting"),
        b_interface_class: class,
        b_interface_sub_class: subclass,
        b_interface_protocol: read_sysfs_hex(&iface_path, "bInterfaceProtocol"),
        description: read_sysfs_attr(&iface_path, "interface"),
        class_name: interface_class_name(class, subclass),
        endpoints: read_endpoints(&iface_path),
        raw_descriptor: vec![],
      });
    }
  }

  interfaces.sort_by_key(|i| i.b_interface_number);
  interfaces
}

pub fn read_device(sysfs_path: &Path) -> Result<NativeUSBDevice> {
  let bus = read_sysfs_u32(sysfs_path, "busnum");
  let dev = read_sysfs_u32(sysfs_path, "devnum");

  if bus == 0 && dev == 0 {
    return Err(napi::Error::from_reason("Not a valid USB device"));
  }

  let id_vendor = read_sysfs_hex(sysfs_path, "idVendor");
  let id_product = read_sysfs_hex(sysfs_path, "idProduct");
  let bcd_usb_raw = read_sysfs_hex(sysfs_path, "version");
  let bcd_device_raw = read_sysfs_hex(sysfs_path, "bcdDevice");
  let speed_str = read_sysfs_attr(sysfs_path, "speed").unwrap_or_default();
  let (speed, speed_mbps) = parse_speed(&speed_str);

  let bcd_usb = read_sysfs_attr(sysfs_path, "version")
    .unwrap_or_else(|| "0.00".into())
    .trim()
    .to_string();

  let raw = read_raw_descriptor(sysfs_path);

  // Read configuration
  let b_max_power_str = read_sysfs_attr(sysfs_path, "bMaxPower").unwrap_or_default();
  let b_max_power: u32 = b_max_power_str
    .trim_end_matches("mA")
    .trim()
    .parse()
    .unwrap_or(0);

  let configurations = vec![NativeConfiguration {
    b_configuration_value: read_sysfs_u32(sysfs_path, "bConfigurationValue"),
    bm_attributes: read_sysfs_hex(sysfs_path, "bmAttributes"),
    b_max_power: b_max_power,
    description: read_sysfs_attr(sysfs_path, "configuration"),
    interfaces: read_interfaces(sysfs_path, bus),
    raw_descriptor: vec![],
  }];

  let sysfs_str = sysfs_path.to_string_lossy().to_string();

  Ok(NativeUSBDevice {
    bus_number: bus,
    device_number: dev,
    port_numbers: parse_port_numbers(&sysfs_str),
    sysfs_path: sysfs_str,
    descriptor: NativeDeviceDescriptor {
      b_length: 18,
      b_descriptor_type: 1,
      bcd_usb,
      b_device_class: read_sysfs_hex(sysfs_path, "bDeviceClass"),
      b_device_sub_class: read_sysfs_hex(sysfs_path, "bDeviceSubClass"),
      b_device_protocol: read_sysfs_hex(sysfs_path, "bDeviceProtocol"),
      b_max_packet_size0: read_sysfs_u32(sysfs_path, "bMaxPacketSize0"),
      id_vendor,
      id_product,
      bcd_device: read_sysfs_attr(sysfs_path, "bcdDevice").unwrap_or_else(|| "0.00".into()),
      i_manufacturer: read_sysfs_u32(sysfs_path, "iManufacturer"),
      i_product: read_sysfs_u32(sysfs_path, "iProduct"),
      i_serial_number: read_sysfs_u32(sysfs_path, "iSerialNumber"),
      b_num_configurations: read_sysfs_u32(sysfs_path, "bNumConfigurations"),
    },
    manufacturer: read_sysfs_attr(sysfs_path, "manufacturer"),
    product: read_sysfs_attr(sysfs_path, "product"),
    serial_number: read_sysfs_attr(sysfs_path, "serial"),
    speed,
    speed_mbps,
    configurations,
    raw_descriptor: raw,
  })
}

pub fn enumerate_devices() -> Vec<NativeUSBDevice> {
  let usb_devices_path = PathBuf::from("/sys/bus/usb/devices");
  let mut devices = vec![];

  let entries = match fs::read_dir(&usb_devices_path) {
    Ok(e) => e,
    Err(_) => return devices,
  };

  for entry in entries.flatten() {
    let path = entry.path();
    // Skip interfaces (contain ":"), only read device dirs
    let name = entry.file_name().to_string_lossy().to_string();
    if name.contains(':') {
      continue;
    }

    // Must have a busnum file to be a USB device
    if !path.join("busnum").exists() {
      continue;
    }

    if let Ok(device) = read_device(&path) {
      devices.push(device);
    }
  }

  devices.sort_by(|a, b| {
    a.bus_number
      .cmp(&b.bus_number)
      .then(a.device_number.cmp(&b.device_number))
  });
  devices
}
```

```rust
// native/src/descriptor.rs
use napi::bindgen_prelude::*;
use serde::{Deserialize, Serialize};

#[derive(Debug, Clone, Serialize, Deserialize)]
#[napi(object)]
pub struct DescriptorField {
  pub name: String,
  pub offset: u32,
  pub size: u32,
  pub raw_bytes: Vec<u32>,
  pub value: u32,
  pub description: String,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
#[napi(object)]
pub struct ParsedDescriptor {
  pub descriptor_type: String,
  pub fields: Vec<DescriptorField>,
  pub raw_bytes: Vec<u32>,
  pub total_length: u32,
}

fn read_u8(data: &[u8], offset: usize) -> u32 {
  *data.get(offset).unwrap_or(&0) as u32
}

fn read_u16_le(data: &[u8], offset: usize) -> u32 {
  let lo = *data.get(offset).unwrap_or(&0) as u32;
  let hi = *data.get(offset + 1).unwrap_or(&0) as u32;
  (hi << 8) | lo
}

fn format_bcd(val: u32) -> String {
  format!("{}.{:02}", (val >> 8) & 0xFF, val & 0xFF)
}

fn descriptor_type_name(t: u32) -> &'static str {
  match t {
    1 => "DEVICE",
    2 => "CONFIGURATION",
    4 => "INTERFACE",
    5 => "ENDPOINT",
    6 => "DEVICE_QUALIFIER",
    3 => "STRING",
    _ => "UNKNOWN",
  }
}

fn device_class_name(class: u32) -> String {
  match class {
    0x00 => "Per-Interface".into(),
    0x01 => "Audio".into(),
    0x02 => "CDC Communications".into(),
    0x03 => "HID".into(),
    0x05 => "Physical".into(),
    0x06 => "Image".into(),
    0x07 => "Printer".into(),
    0x08 => "Mass Storage".into(),
    0x09 => "Hub".into(),
    0x0A => "CDC Data".into(),
    0x0E => "Video".into(),
    0xDC => "Diagnostic".into(),
    0xE0 => "Wireless Controller".into(),
    0xEF => "Miscellaneous".into(),
    0xFE => "Application Specific".into(),
    0xFF => "Vendor Specific".into(),
    _ => format!("Unknown (0x{:02X})", class),
  }
}

fn transfer_type_name(attrs: u32) -> &'static str {
  match attrs & 0x03 {
    0 => "Control",
    1 => "Isochronous",
    2 => "Bulk",
    3 => "Interrupt",
    _ => "Unknown",
  }
}

    value: max_pkt,
    description: format!("{} bytes", max_pkt & 0x7FF),
  });

  fields.push(DescriptorField {
    name: "bInterval".into(), offset: 6, size: 1,
    raw_bytes: vec![raw.get(6).copied().unwrap_or(0)],
    value: read_u8(data, 6),
    description: format!("{} ms", read_u8(data, 6)),
  });

  ParsedDescriptor {
    descriptor_type: "ENDPOINT".into(),
    fields,
    raw_bytes: raw[..7.min(raw.len())].to_vec(),
    total_length: 7,
  }
}

#[napi]
pub fn parse_descriptor(raw_bytes: Vec<u32>, descriptor_type: String) -> ParsedDescriptor {
  let data: Vec<u8> = raw_bytes.iter().map(|b| *b as u8).collect();
  match descriptor_type.as_str() {
    "device" => parse_device_descriptor(&data),
    "configuration" => parse_configuration_descriptor(&data),
    "interface" => parse_interface_descriptor(&data),
    "endpoint" => parse_endpoint_descriptor(&data),
    _ => parse_device_descriptor(&data),
  }
}
```

```rust
use napi::bindgen_prelude::*;
use napi::threadsafe_function::{
  ErrorStrategy, ThreadsafeFunction, ThreadsafeFunctionCallMode,
};
use serde::{Deserialize, Serialize};
use std::sync::atomic::{AtomicBool, Ordering};
use std::sync::Arc;
use std::thread;

use crate::device::{read_device, NativeUSBDevice};

#[derive(Debug, Clone, Serialize, Deserialize)]
#[napi(object)]
pub struct MonitorEvent {
  pub action: String, // "add" | "remove"
  pub sysfs_path: String,
  pub device: Option<NativeUSBDevice>,
}

#[napi]
pub struct MonitorHandle {
  running: Arc<AtomicBool>,
}

#[napi]
impl MonitorHandle {
  #[napi]
  pub fn stop(&self) {
    self.running.store(false, Ordering::SeqCst);
  }
}

#[napi]
pub fn start_monitor(
  callback: ThreadsafeFunction<MonitorEvent, ErrorStrategy::Fatal>,
) -> Result<MonitorHandle> {
  let running = Arc::new(AtomicBool::new(true));
  let running_clone = running.clone();

  thread::spawn(move || {
    let socket = match udev::MonitorBuilder::new() {
      Ok(builder) => match builder
        .match_subsystem("usb")
        .and_then(|b| b.listen())
      {
        Ok(s) => s,
        Err(e) => {
          eprintln!("Failed to create udev monitor: {}", e);
          return;
        }
      },
      Err(e) => {
        eprintln!("Failed to create udev monitor builder: {}", e);
        return;
      }
    };

    let poll_fd = match socket.as_raw_fd() {
      fd => fd,
    };

    use std::os::fd::AsRawFd;

    while running_clone.load(Ordering::SeqCst) {
      // Poll with 500ms timeout to check running flag
      let mut poll_fds = [nix::poll::PollFd::new(
        unsafe { std::os::fd::BorrowedFd::borrow_raw(socket.as_raw_fd()) },
        nix::poll::PollFlags::POLLIN,
      )];

      match nix::poll::poll(&mut poll_fds, nix::poll::PollTimeout::from(500u16)) {
        Ok(n) if n > 0 => {}
        _ => continue,
      }

      if let Some(event) = socket.iter().next() {
        let action = event
          .action()
          .and_then(|a| a.to_str().ok())
          .unwrap_or("")
          .to_string();

        if action != "add" && action != "remove" {
          continue;
        }

        let sysfs_path = event.syspath().to_string_lossy().to_string();

        // Skip interface events (contain ":")
        if sysfs_path.split('/').last().unwrap_or("").contains(':') {
          continue;
        }

        let device = if action == "add" {
          // Small delay for sysfs to populate
          thread::sleep(std::time::Duration::from_millis(200));
          read_device(&std::path::PathBuf::from(&sysfs_path)).ok()
        } else {
          None
        };

        let evt = MonitorEvent {
          action,
          sysfs_path,
          device,
        };

        callback.call(evt, ThreadsafeFunctionCallMode::NonBlocking);
      }
    }
  });

  Ok(MonitorHandle { running })
}
```

```rust
// native/src/usb_ids.rs
use napi::bindgen_prelude::*;
use serde::{Deserialize, Serialize};
use std::collections::HashMap;
use std::fs;

#[derive(Debug, Clone, Serialize, Deserialize)]
#[napi(object)]
pub struct VendorEntry {
  pub id: u32,
  pub name: String,
  pub products: Vec<ProductEntry>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
#[napi(object)]
pub struct ProductEntry {
  pub id: u32,
  pub name: String,
}

#[napi]
pub fn parse_usb_ids(file_path: Option<String>) -> Vec<VendorEntry> {
  let path = file_path.unwrap_or_else(|| "/usr/share/hwdata/usb.ids".into());
  let content = match fs::read_to_string(&path) {
    Ok(c) => c,
    Err(_) => {
      // Try alternative paths
      let alt_paths = [
        "/usr/share/misc/usb.ids",
        "/usr/share/usb.ids",
        "/var/lib/usbutils/usb.ids",
      ];
      let mut found = String::new();
      for alt in &alt_paths {
        if let Ok(c) = fs::read_to_string(alt) {
          found = c;
          break;
        }
      }
      if found.is_empty() {
        return vec![];
      }
      found
    }
  };

  let mut vendors = vec![];
  let mut current_vendor: Option<VendorEntry> = None;

  for line in content.lines() {
    // Skip comments and empty lines
    if line.starts_with('#') || line.is_empty() {
      continue;
    }

    // Vendor line: starts with hex ID (no leading whitespace)
    if !line.starts_with('\t') && !line.starts_with(' ') {
      // Save previous vendor
      if let Some(v) = current_vendor.take() {
        vendors.push(v);
      }

      let parts: Vec<&str> = line.splitn(2, "  ").collect();
      if parts.len() == 2 {
        if let Ok(id) = u32::from_str_radix(parts[0].trim(), 16) {
          current_vendor = Some(VendorEntry {
            id,
            name: parts[1].trim().to_string(),
            products: vec![],
          });
        }
      }
      continue;
    }

    // Product line: starts with single tab
    if line.starts_with('\t') && !line.starts_with("\t\t") {
      if let Some(ref mut vendor) = current_vendor {
        let trimmed = line.trim();
        let parts: Vec<&str> = trimmed.splitn(2, "  ").collect();
        if parts.len() == 2 {
          if let Ok(id) = u32::from_str_radix(parts[0].trim(), 16) {
            vendor.products.push(ProductEntry {
              id,
              name: parts[1].trim().to_string(),
            });
          }
        }
      }
    }
  }

  // Don't forget the last vendor
  if let Some(v) = current_vendor {
    vendors.push(v);
  }

  vendors
}

#[napi]
pub fn lookup_vendor(vendors: Vec<VendorEntry>, vid: u32) -> Option<VendorEntry> {
  vendors.into_iter().find(|v| v.id == vid)
}

#[napi]
pub fn lookup_product(vendors: Vec<VendorEntry>, vid: u32, pid: u32) -> Option<ProductEntry> {
  vendors
    .into_iter()
    .find(|v| v.id == vid)
    .and_then(|v| v.products.into_iter().find(|p| p.id == pid))
}
```

```typescript
function handleMessage(msg: MainToWorkerMessage): void {
  switch (msg.type) {
    case 'list-devices': {
      try {
        const devices = native.listDevices()
        postMessage({ type: 'device-list', devices })
      } catch (e) {
        postMessage({ type: 'error', message: `Failed to list devices: ${e}` })
      }
      break
    }

    case 'start-monitor': {
      if (monitorHandle) return
      try {
        monitorHandle = native.startMonitor((event: {
          action: string
          sysfsPath: string
          device: unknown
        }) => {
          postMessage({
            type: 'device-event',
            event: {
              action: event.action as 'add' | 'remove',
              sysfsPath: event.sysfsPath,
              device: event.device as any
            }
          })
        })
      } catch (e) {
        postMessage({ type: 'error', message: `Failed to start monitor: ${e}` })
      }
      break
    }

    case 'stop-monitor': {
      if (monitorHandle) {
        monitorHandle.stop()
        monitorHandle = null
      }
      break
    }

    case 'get-descriptor': {
      try {
        const device = native.getDeviceDescriptor(msg.sysfsPath)
        postMessage({ type: 'device-list', devices: [device] })
      } catch (e) {
        postMessage({ type: 'error', message: `Failed to get descriptor: ${e}` })
      }
      break
    }

    case 'get-raw-descriptor': {
      try {
        const raw = native.getRawDescriptor(msg.sysfsPath)
        postMessage({ type: 'device-list', devices: raw })
      } catch (e) {
        postMessage({ type: 'error', message: `Failed to get raw descriptor: ${e}` })
      }
      break
    }
  }
```

```typescript
// src/main/worker-manager.ts
import { Worker } from 'worker_threads'
import { join } from 'path'
import type { WorkerMessage, MainToWorkerMessage } from '../shared/types'

type WorkerEventHandler = (msg: WorkerMessage) => void

let worker: Worker | null = null
let handlers: WorkerEventHandler[] = []

export function startWorker(): void {
  if (worker) return

  worker = new Worker(join(__dirname, 'usb-worker.js'))

  worker.on('message', (msg: WorkerMessage) => {
    for (const handler of handlers) {
      handler(msg)
    }
  })

  worker.on('error', (err) => {
    console.error('USB Worker error:', err)
  })

  worker.on('exit', (code) => {
    console.log('USB Worker exited with code:', code)
    worker = null
  })
}

export function stopWorker(): void {
  if (worker) {
    sendToWorker({ type: 'stop-monitor' })
    worker.terminate()
    worker = null
  }
}

export function sendToWorker(msg: MainToWorkerMessage): void {
  worker?.postMessage(msg)
}

export function onWorkerMessage(handler: WorkerEventHandler): () => void {
  handlers.push(handler)
  return () => {
    handlers = handlers.filter((h) => h !== handler)
  }
}
```

```typescript
export function setupIPC(mainWindow: BrowserWindow): void {
  // Load USB IDs on startup
  loadUsbIds()

  // Forward worker messages to renderer
  onWorkerMessage((msg: WorkerMessage) => {
    switch (msg.type) {
      case 'device-list': {
        currentDevices = msg.devices
        // Enrich devices with vendor info
        for (const dev of currentDevices) {
          dev.vendor = lookupVendorById(dev.descriptor.idVendor)
          dev.productInfo = lookupProductById(dev.descriptor.idVendor, dev.descriptor.idProduct)
        }
        mainWindow.webContents.send('usb:device-list', currentDevices)
        break
      }
      case 'device-event': {
        const { action, sysfsPath, device } = msg.event
        if (device) {
          device.vendor = lookupVendorById(device.descriptor.idVendor)
          device.productInfo = lookupProductById(
            device.descriptor.idVendor,
            device.descriptor.idProduct
          )
        }

          id: uuidv4(),
          timestamp: new Date().toISOString(),
          type: action === 'add' ? 'connect' : 'disconnect',
          busNumber: device?.busNumber ?? 0,
          deviceNumber: device?.deviceNumber ?? 0,
          sysfsPath,
          device: device ?? null,
          summary: `${device?.product ?? 'Unknown device'} ${action === 'add' ? 'connected' : 'disconnected'} on Bus ${String(device?.busNumber ?? '?').padStart(3, '0')}`
        }

        insertEvent(event)
        mainWindow.webContents.send('usb:device-event', event)

        // Refresh device list
        sendToWorker({ type: 'list-devices' })
        break
```

```typescript
// src/main/database.ts
import Database from 'better-sqlite3'
import { app } from 'electron'
import { join } from 'path'
import type { USBEvent } from '../shared/types'

const SCHEMA_VERSION = 1

let db: Database.Database | null = null

export function initDatabase(): void {
  const dbPath = join(app.getPath('userData'), 'usbprobe.db')
  db = new Database(dbPath)

  db.pragma('journal_mode = WAL')

  db.exec(`
    CREATE TABLE IF NOT EXISTS metadata (
      key TEXT PRIMARY KEY,
      value TEXT NOT NULL
    );

    CREATE TABLE IF NOT EXISTS events (
      id TEXT PRIMARY KEY,
      timestamp TEXT NOT NULL,
      type TEXT NOT NULL,
      bus_number INTEGER NOT NULL,
      device_number INTEGER NOT NULL,
      sysfs_path TEXT NOT NULL,
      device_snapshot TEXT,
      summary TEXT NOT NULL
    );

    CREATE INDEX IF NOT EXISTS idx_events_timestamp ON events(timestamp DESC);
    CREATE INDEX IF NOT EXISTS idx_events_type ON events(type);
  `)

  // Set schema version
  const stmt = db.prepare('INSERT OR REPLACE INTO metadata (key, value) VALUES (?, ?)')
  stmt.run('schema_version', String(SCHEMA_VERSION))
  stmt.run('created_at', new Date().toISOString())
}

export function closeDatabase(): void {
  db?.close()
  db = null
}

export function insertEvent(event: USBEvent): void {
  if (!db) return
  const stmt = db.prepare(`
    INSERT INTO events (id, timestamp, type, bus_number, device_number, sysfs_path, device_snapshot, summary)
    VALUES (?, ?, ?, ?, ?, ?, ?, ?)
  `)
  stmt.run(
    event.id,
    event.timestamp,
    event.type,
    event.busNumber,
    event.deviceNumber,
    event.sysfsPath,
    event.device ? JSON.stringify(event.device) : null,
    event.summary
  )
}

export function getEvents(limit = 100, offset = 0): USBEvent[] {
  if (!db) return []
  const rows = db
    .prepare('SELECT * FROM events ORDER BY timestamp DESC LIMIT ? OFFSET ?')
    .all(limit, offset) as {
    id: string
    timestamp: string
    type: string
    bus_number: number
    device_number: number
    sysfs_path: string
    device_snapshot: string | null
    summary: string
  }[]

  return rows.map((row) => ({
    id: row.id,
    timestamp: row.timestamp,
    type: row.type as 'connect' | 'disconnect',
    busNumber: row.bus_number,
    deviceNumber: row.device_number,
    sysfsPath: row.sysfs_path,
    device: row.device_snapshot ? JSON.parse(row.device_snapshot) : null,
    summary: row.summary
  }))
}

export function getEventCount(): number {
  if (!db) return 0
  const row = db.prepare('SELECT COUNT(*) as count FROM events').get() as { count: number }
  return row.count
}

export function searchEvents(query: string): USBEvent[] {
  if (!db) return []
  const rows = db
    .prepare(
      `SELECT * FROM events
       WHERE summary LIKE ? OR device_snapshot LIKE ?
       ORDER BY timestamp DESC LIMIT 200`
    )
    .all(`%${query}%`, `%${query}%`) as {
    id: string
    timestamp: string
    type: string
    bus_number: number
    device_number: number
    sysfs_path: string
    device_snapshot: string | null
    summary: string
  }[]

    id: row.id,
```

```typescript
// src/main/udev-rules.ts
import { readFileSync, readdirSync, existsSync } from 'fs'
import { execSync } from 'child_process'
import { v4 as uuidv4 } from 'uuid'
import { writeFile } from 'fs/promises'
import { join } from 'path'
import { app } from 'electron'
import type { UdevRule } from '../shared/types'

const RULES_DIR = '/etc/udev/rules.d'
const RULES_PREFIX = '99-usbprobe'

export function listUdevRules(): UdevRule[] {
  const rules: UdevRule[] = []
  if (!existsSync(RULES_DIR)) return rules

  const files = readdirSync(RULES_DIR).filter((f) => f.startsWith(RULES_PREFIX))

  for (const file of files) {
    const content = readFileSync(join(RULES_DIR, file), 'utf-8')
    for (const line of content.split('\n')) {
      const trimmed = line.trim()
      if (!trimmed || trimmed.startsWith('#')) continue

      const rule: UdevRule = {
        id: uuidv4(),
        filename: file,
        rawRule: trimmed,
        managedByUSBProbe: true
      }

      // Parse match conditions
      const vendorMatch = trimmed.match(/ATTR\{idVendor\}=="([0-9a-fA-F]+)"/)
      if (vendorMatch) rule.matchVendor = parseInt(vendorMatch[1], 16)

      const productMatch = trimmed.match(/ATTR\{idProduct\}=="([0-9a-fA-F]+)"/)
      if (productMatch) rule.matchProduct = parseInt(productMatch[1], 16)

      const modeMatch = trimmed.match(/MODE="([^"]+)"/)
      if (modeMatch) rule.actionMode = modeMatch[1]

      const groupMatch = trimmed.match(/GROUP="([^"]+)"/)
      if (groupMatch) rule.actionGroup = groupMatch[1]

      rules.push(rule)
    }
  }
  return rules
}

export async function addUdevRule(rule: Partial<UdevRule>): Promise<void> {
  let ruleStr = 'SUBSYSTEM=="usb"'
  if (rule.matchVendor !== undefined) {
    ruleStr += `, ATTR{idVendor}=="${rule.matchVendor.toString(16).padStart(4, '0')}"`
  }
  if (rule.matchProduct !== undefined) {
    ruleStr += `, ATTR{idProduct}=="${rule.matchProduct.toString(16).padStart(4, '0')}"`
  }
  if (rule.actionMode) ruleStr += `, MODE="${rule.actionMode}"`
  if (rule.actionGroup) ruleStr += `, GROUP="${rule.actionGroup}"`
  if (rule.actionSymlink) ruleStr += `, SYMLINK+="${rule.actionSymlink}"`
  if (rule.actionRun) ruleStr += `, RUN+="${rule.actionRun}"`

  const filename = `${RULES_PREFIX}-custom.rules`
  const tempPath = join(app.getPath('temp'), filename)
  const targetPath = join(RULES_DIR, filename)

  // Read existing rules and append
  let existing = ''
  try {
    existing = readFileSync(targetPath, 'utf-8')
  } catch {
    existing = '# USBProbe custom udev rules\n'
  }

  await writeFile(tempPath, existing + ruleStr + '\n', 'utf-8')

  try {
    execSync(`pkexec cp "${tempPath}" "${targetPath}"`)
    execSync('pkexec udevadm control --reload-rules')
    execSync('pkexec udevadm trigger')
  } catch (e) {
    throw new Error(`Failed to install udev rule: ${e}`)
  }
}

export async function removeUdevRule(id: string): Promise<void> {
  const rules = listUdevRules()
  const rule = rules.find((r) => r.id === id)
  if (!rule) return

  const filePath = join(RULES_DIR, rule.filename)
  const content = readFileSync(filePath, 'utf-8')
  const newContent = content
    .split('\n')
    .filter((line) => line.trim() !== rule.rawRule)
    .join('\n')

  const tempPath = join(app.getPath('temp'), rule.filename)
  await writeFile(tempPath, newContent, 'utf-8')

  try {
    execSync(`pkexec cp "${tempPath}" "${filePath}"`)
    execSync('pkexec udevadm control --reload-rules')
  } catch (e) {
    throw new Error(`Failed to remove udev rule: ${e}`)
  }
}

export async function installPermissionRule(): Promise<boolean> {
  const ruleContent = `# USBProbe USB device access
SUBSYSTEM=="usb", MODE="0664", GROUP="plugdev"
`
  const filename = `${RULES_PREFIX}-permissions.rules`
  const tempPath = join(app.getPath('temp'), filename)
  const targetPath = join(RULES_DIR, filename)

  await writeFile(tempPath, ruleContent, 'utf-8')

  try {
    execSync(`pkexec cp "${tempPath}" "${targetPath}"`)
    execSync('pkexec udevadm control --reload-rules')
    execSync('pkexec udevadm trigger')
    return true
  } catch {
    return false
  }
}
```

```typescript
// src/renderer/hooks/use-usb-api.ts
import { useEffect } from 'react'
import { useDeviceStore } from '../stores/device-store'
import type { USBProbeAPI, USBDevice, USBEvent } from '../../shared/types'

declare global {
  interface Window {
    usbProbe: USBProbeAPI
    usbProbeEvents: {
      onDeviceListUpdate: (cb: (devices: USBDevice[]) => void) => () => void
      onError: (cb: (message: string) => void) => () => void
    }
  }
}

export const api = window.usbProbe

export function useDeviceListSync(): void {
  const setDevices = useDeviceStore((s) => s.setDevices)
  const addEvent = useDeviceStore((s) => s.addEvent)

  useEffect(() => {
    // Initial load
    api.listDevices().then(setDevices)
    api.getEvents(50).then((events) => useDeviceStore.getState().setEvents(events))

    // Subscribe to updates
    const unsubDevices = window.usbProbeEvents.onDeviceListUpdate(setDevices)
    const unsubEvents = api.onDeviceEvent((event) => addEvent(event))

    return () => {
      unsubDevices()
      unsubEvents()
    }
  }, [setDevices, addEvent])
}
```

```tsx
import { useState, useMemo } from 'react'
import { useTranslation } from 'react-i18next'
import { useDeviceStore } from '../stores/device-store'
import { SearchFilter } from './SearchFilter'
import { TreeNode } from './TreeNode'
import type { USBDevice } from '../../shared/types'

interface BusGroup {
  busNumber: number
  devices: USBDevice[]
}

export function DeviceTreePanel(): React.ReactElement {
  const { t } = useTranslation('devices')
  const devices = useDeviceStore((s) => s.devices)
  const [filter, setFilter] = useState('')

  const filteredDevices = useMemo(() => {
    if (!filter) return devices
    const lc = filter.toLowerCase()
    return devices.filter(
      (d) =>
        (d.product ?? '').toLowerCase().includes(lc) ||
        (d.manufacturer ?? '').toLowerCase().includes(lc) ||
        (d.vendor?.name ?? '').toLowerCase().includes(lc) ||
        `${d.descriptor.idVendor.toString(16)}:${d.descriptor.idProduct.toString(16)}`.includes(lc)
    )
  }, [devices, filter])

  const busGroups = useMemo(() => {
    const groups = new Map<number, USBDevice[]>()
    for (const dev of filteredDevices) {
      const list = groups.get(dev.busNumber) ?? []
      list.push(dev)
      groups.set(dev.busNumber, list)
    }
    return Array.from(groups.entries())
      .sort(([a], [b]) => a - b)
      .map(([busNumber, devs]): BusGroup => ({ busNumber, devices: devs }))
  }, [filteredDevices])

  const connectedCount = devices.length

  return (
    <div
      style={{
        width: 250,
        background: 'var(--bg-primary)',
        borderRight: '1px solid var(--border)',
        padding: 10,
        overflowY: 'auto',
        flexShrink: 0
      }}
    >
      <div
        style={{
          display: 'flex',
          alignItems: 'center',
          justifyContent: 'space-between',
          marginBottom: 10
        }}
      >
        <span className="label">{t('tree.title')}</span>
        <span style={{ color: 'var(--success)', fontSize: 10 }}>● {connectedCount}</span>
      </div>

      <SearchFilter value={filter} onChange={setFilter} />

      <div style={{ fontFamily: 'monospace', fontSize: 11, lineHeight: 1.8 }}>
        {busGroups.map((group) => (
          <TreeNode
            key={group.busNumber}
            label={`${t('tree.bus')} ${String(group.busNumber).padStart(3, '0')}`}
            icon="mdi:bus"
            iconColor="var(--accent)"
          >
            {group.devices.map((dev) => {
              const isHub = dev.descriptor.bDeviceClass === 0x09
              return (
                <TreeNode
                  key={dev.sysfsPath}
                  label={dev.product ?? dev.vendor?.name ?? `Device ${dev.deviceNumber}`}
                  icon={isHub ? 'mdi:hub' : 'mdi:usb'}
                  iconColor={isHub ? 'var(--ctp-yellow)' : 'var(--success)'}
                  sysfsPath={dev.sysfsPath}
                  device={dev}
                />
              )
            })}
          </TreeNode>
        ))}
```

```tsx
import type { USBDevice } from '../../shared/types'

type Base = 'hex' | 'dec' | 'bin' | 'oct'

interface DescriptorFieldInfo {
  name: string
  offset: number
  size: number
  colorIndex: number
}

// Exported for testing
export function formatByte(byte: number, base: Base): string {
  switch (base) {
    case 'hex': return byte.toString(16).toUpperCase().padStart(2, '0')
    case 'dec': return byte.toString(10).padStart(3, '0')
    case 'bin': return byte.toString(2).padStart(8, '0')
    case 'oct': return byte.toString(8).padStart(3, '0')
  }
}

export function formatRow(bytes: number[], base: Base): string[] {
  return bytes.map((b) => formatByte(b, base))
}

  'var(--hex-field-0)', 'var(--hex-field-1)', 'var(--hex-field-2)',
  'var(--hex-field-3)', 'var(--hex-field-4)', 'var(--hex-field-5)',

  'var(--hex-field-9)'
]
```

```tsx
  const { t } = useTranslation()
  const [rules, setRules] = useState<UdevRule[]>([])
  const [showAdd, setShowAdd] = useState(false)
  const [vid, setVid] = useState('')
  const [pid, setPid] = useState('')
  const [mode, setMode] = useState('0664')
  const [group, setGroup] = useState('plugdev')

  const loadRules = async (): Promise<void> => {
    const r = await api.listUdevRules()
    setRules(r)
  }

  useEffect(() => { loadRules() }, [])

  const handleAdd = async (): Promise<void> => {
    await api.addUdevRule({
      matchVendor: vid ? parseInt(vid, 16) : undefined,
      matchProduct: pid ? parseInt(pid, 16) : undefined,
      actionMode: mode || undefined,
      actionGroup: group || undefined

    setVid('')
```
